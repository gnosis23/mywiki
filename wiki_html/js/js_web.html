一些普通应用 

  // <input type="text" name="t">通过如下形式获取对话框的值
  var name = document.myForm.t.value;
  // <select name="s" [selected]>下拉菜单
  document.myForm.s.value;
  // <div>向div中插入值
  document.getElementById(divName).innerHTML += name;
  // <input type="radio" name="choice" value="x" [checked] /> * n
  // 通过如下形式获取单选框的选项
  for(var i = 0; i < document.myForm.choice.length; i++) 
  {
    if( document.myForm.choice[i].checked ){
      document.myForm.choice[i].value;
    }
  }
  // <input type="checkbox" name="ck" />多选框 
  // 和单选差不多啊
