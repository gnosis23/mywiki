循环数组 

<?php
// range
$numbers = range(1, 10, 3);
print_r($numbers); //1, 4, 7 , 10 
echo "<br/>";

$prices = array('Tires' => 100, 'Oil'=>10, 'Spark'=>4);

// 循环
foreach ($prices as $key => $value) {
  echo $key." - ".$value."<br/>";
}

// each会记录数组当前位置，多次使用要reset
reset($prices);
while($element = each($prices)){
  echo $element['key'];
  echo " -- ";
  echo $element['value'];
  echo "<br />";
}

// list
reset($prices);
while( list($product, $price) = each($prices)){
  echo "$product --- $prices<br />";
}
?>
