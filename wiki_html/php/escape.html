mysqli_real_escape_string函数 

  $conn = db_connect();

  // 预防sql注入

  $query = sprintf("insert into user values('%s', sha1('%s'), '%s')",
                    $conn->real_escape_string($username),
                    $conn->real_escape_string($passwd),
                    $conn->real_escape_string($email));
  $result = $conn->query($query);
